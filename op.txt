-- ==========================================
-- 👁️ ESP SYSTEM — Event-Driven (ใหม่)
-- แก้ปัญหา: เปิด/ปิดไม่ค้าง, respawn เจอทันที,
--           player เข้า-ออกแมพเจอทันที, ไม่ทิ้งซาก
-- ==========================================
local _espStore = {}  -- [player] = { hl }
local _espConns = {}  -- [player] = { charAddedConn, charRemovingConn }

local function espDestroyOne(player)
    local d = _espStore[player]
    if not d then return end
    pcall(function() d.hl:Destroy() end)
    _espStore[player] = nil
end

local function espBuildOne(player, char)
    espDestroyOne(player)   -- ล้างของเก่าก่อนเสมอ ไม่ซ้อน
    if not char then return end
    local root = char:WaitForChild("HumanoidRootPart", 8)
    if not root then return end

    local hl = Instance.new("Highlight")
    hl.Name                = "ChibaHighlight"
    hl.Adornee             = char
    hl.DepthMode           = Enum.HighlightDepthMode.AlwaysOnTop
    hl.FillColor           = Color3.fromRGB(255, 50, 50)
    hl.FillTransparency    = 0.5
    hl.OutlineColor        = Color3.fromRGB(100, 0, 0)
    hl.OutlineTransparency = 0
    hl.Enabled             = true
    hl.Parent              = char

    _espStore[player] = { hl = hl }
end

local function espSetupPlayer(p)
    if p == lp or _espConns[p] then return end  -- guard: ไม่ setup ซ้ำ

    -- ฟัง event ตลอด ไม่ว่า ESP จะเปิดหรือปิด
    local c1 = p.CharacterAdded:Connect(function(char)
        if ESPState then task.defer(espBuildOne, p, char) end
    end)
    local c2 = p.CharacterRemoving:Connect(function()
        espDestroyOne(p)
    end)
    _espConns[p] = { c1, c2 }

    -- ถ้า ESP เปิดอยู่แล้วและ player มี char แล้ว → สร้างเลย
    if ESPState and p.Character then
        espBuildOne(p, p.Character)
    end
end

local function espTeardownPlayer(p)
    local cs = _espConns[p]
    if cs then
        for _, c in ipairs(cs) do c:Disconnect() end
        _espConns[p] = nil
    end
    espDestroyOne(p)
end

-- ── PUBLIC: ถูกเรียกจาก Toggle ด้านบน ─────────────────────
function espEnable()
    -- สร้าง ESP ให้ทุกคนที่อยู่ในเกมตอนนี้
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= lp and p.Character then
            espBuildOne(p, p.Character)
        end
    end
end

function espDisable()
    -- ลบทุกอย่างออกให้หมด ไม่ทิ้งซากไว้เลย
    for p in pairs(_espStore) do
        espDestroyOne(p)
    end
    table.clear(_espStore)
end

-- ── Event: player เข้า/ออกเกม ──────────────────────────────
Players.PlayerAdded:Connect(espSetupPlayer)
Players.PlayerRemoving:Connect(espTeardownPlayer)

-- setup ผู้เล่นที่อยู่ในเกมอยู่แล้ว (กรณีสคริปต์รันทีหลัง)
for _, p in ipairs(Players:GetPlayers()) do
    espSetupPlayer(p)
end

-- ── Throttled update: อัปเดตสีตาม context (~8fps ประหยัด CPU มือถือ) ──
local _espTick = 0
RS.Heartbeat:Connect(function()
    if not ESPState then return end  -- หยุดทันทีถ้าปิด ESP
    local now = tick()
    if now - _espTick < 0.12 then return end
    _espTick = now

    for player, d in pairs(_espStore) do
        -- player ออกจากเกมไปแล้ว
        if not player.Parent then espDestroyOne(player); continue end

        local char = player.Character
        if not char or not char.Parent then
            d.hl.Enabled = false; continue
        end

        local hum = char:FindFirstChildOfClass("Humanoid")
        if not hum or hum.Health <= 0 then
            d.hl.Enabled = false; continue
        end

        -- ป้องกัน Highlight หลุด parent หลัง respawn
        if d.hl.Parent ~= char then
            d.hl.Adornee = char
            d.hl.Parent  = char
        end

        d.hl.Enabled = true

        -- สีตาม context (เหมือนระบบเดิมทุกอย่าง)
        if player == ActiveTarget then
            d.hl.FillColor    = Color3.fromHSV(tick() % 5 / 5, 0.7, 1)
            d.hl.OutlineColor = Color3.new(1, 1, 1)
        elseif TeamList[player.Name] then
            d.hl.FillColor    = Color3.fromRGB(0, 150, 255)
            d.hl.OutlineColor = Color3.new(1, 1, 1)
        else
            d.hl.FillColor    = Color3.fromRGB(255, 50, 50)
            d.hl.OutlineColor = Color3.fromRGB(100, 0, 0)
        end
    end
end)

-- ⚙️ 5. COMBAT RENDER STEPPED (GHOST TARGET TOGGLE)
-- ==========================================
RS.RenderStepped:Connect(function()
    ActiveTarget = nil
    if TapLock and TappedTarget and TappedTarget.Character and TappedTarget.Character.Humanoid.Health > 0 then ActiveTarget = TappedTarget
    elseif (AutoNearest or RawLock) and lp.Character and lp.Character:FindFirstChild("HumanoidRootPart") then
        local nearest, minDist = nil, math.huge
        for _, p in pairs(Players:GetPlayers()) do
            if p ~= lp and p.Character and p.Character:FindFirstChild("HumanoidRootPart") and p.Character.Humanoid.Health > 0 and not TeamList[p.Name] then
                local d = (lp.Character.HumanoidRootPart.Position - p.Character.HumanoidRootPart.Position).Magnitude
                if d < minDist then minDist = d; nearest = p end
            end
        end
        ActiveTarget = nearest
    end

    if lp.Character and lp.Character:FindFirstChild("Humanoid") and lp.Character:FindFirstChild("HumanoidRootPart") then
        if ActiveTarget and ActiveTarget.Character and ActiveTarget.Character:FindFirstChild("HumanoidRootPart") then
            lp.Character.Humanoid.AutoRotate = false
            local myHRP = lp.Character.HumanoidRootPart
            local eHRP = ActiveTarget.Character.HumanoidRootPart

            if RawLock then
                myHRP.CFrame = CFrame.new(myHRP.Position, eHRP.Position)
                for _, part in pairs(ActiveTarget.Character:GetChildren()) do
                    if part:IsA("BasePart") and part.Name ~= "HumanoidRootPart" then
                        part.CanCollide = true
                    end
                end
            else
                local dynamicPredict = (currentPing * 0.0016) + 0.1
                local aimPos = eHRP.Position + (eHRP.Velocity * dynamicPredict)
                local lookVector = Vector3.new(aimPos.X - myHRP.Position.X, 0, aimPos.Z - myHRP.Position.Z)
                if lookVector.Magnitude > 0.01 then
                    local targetCFrame = CFrame.new(myHRP.Position, myHRP.Position + lookVector)
                    myHRP.CFrame = myHRP.CFrame:Lerp(targetCFrame, 0.7)
                end
                for _, part in pairs(ActiveTarget.Character:GetChildren()) do
                    if part:IsA("BasePart") and part.Name ~= "HumanoidRootPart" then
                        part.CanCollide = false
                    end
                end
            end
        else
            lp.Character.Humanoid.AutoRotate = true
        end
    end
end)
